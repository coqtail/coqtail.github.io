Compatibility with common operators (continuity at a point)

Lemma continuity_pt_add : ∀ f1 f2 z,
    continuity_pt f1 z → continuity_pt f2 z → continuity_pt (f1 + f2) z.

Lemma continuity_pt_opp : ∀ f z, continuity_pt f z → continuity_pt (- f) z.

Lemma continuity_pt_minus : ∀ f1 f2 (z:C),
    continuity_pt f1 z → continuity_pt f2 z → continuity_pt (f1 - f2) z.

Lemma continuity_pt_mult : ∀ f1 f2 z,
    continuity_pt f1 z → continuity_pt f2 z → continuity_pt (f1 × f2) z.

Lemma continuity_pt_const : ∀ f z, constant f → continuity_pt f z.

Lemma continuity_pt_scal : ∀ f a z,
    continuity_pt f z → continuity_pt (mult_real_fct a f) z.

Lemma continuity_pt_inv : ∀ f z, continuity_pt f z → f z ≠ 0 → continuity_pt (/ f) z.

Lemma div_eq_inv : ∀ f1 f2, (f1 / f2)%F = (f1 × / f2)%F.

Lemma continuity_pt_div : ∀ f1 f2 z,
    continuity_pt f1 z → continuity_pt f2 z → f2 z ≠ 0 →
    continuity_pt (f1 / f2) z.

Lemma continuity_pt_comp : ∀ f1 f2 z,
    continuity_pt f1 z → continuity_pt f2 (f1 z) → continuity_pt (comp f2 f1) z.

Compatibility with common operators (global continuity)

Lemma continuity_plus : ∀ f1 f2,
 continuity f1 → continuity f2 → continuity (f1 + f2).

Lemma continuity_opp : ∀ f, continuity f → continuity (- f).

Lemma continuity_minus : ∀ f1 f2,
 continuity f1 → continuity f2 → continuity (f1 - f2).

Lemma continuity_mult : ∀ f1 f2,
 continuity f1 → continuity f2 → continuity (f1 × f2).

Lemma continuity_const : ∀ f, constant f → continuity f.

Lemma continuity_scal : ∀ f a, continuity f → continuity (mult_real_fct a f).

Lemma continuity_inv : ∀ f, continuity f → (∀ x, f x ≠ 0) → continuity (/ f).

Lemma continuity_div : ∀ f1 f2,
    continuity f1 → continuity f2 → (∀ x, f2 x ≠ 0) → continuity (f1 / f2).

Lemma continuity_comp : ∀ f1 f2,
 continuity f1 → continuity f2 → continuity (comp f2 f1).
