Open Scope R_scope.

Implicit Type r : R.

Lemma Rmin_opp_opp_Rmax : ∀ r1 r2, Rmin (-r1) (-r2) = - (Rmax r1 r2).

Lemma Rmax_opp_opp_Rmin : ∀ r1 r2, Rmax (-r1) (-r2) = - (Rmin r1 r2).

Lemma Rmin_eq_l : ∀ r1 r2, r1 ≤ r2 → Rmin r1 r2 = r1.

Lemma Rmin_eq_r : ∀ r1 r2, r1 ≤ r2 → Rmin r2 r1 = r1.

Lemma Rmax_eq_l : ∀ r1 r2, r2 ≤ r1 → Rmax r1 r2 = r1.

Lemma Rmax_eq_r : ∀ r1 r2, r2 ≤ r1 → Rmax r2 r1 = r1.

Lemma Rabs_eq_compat : ∀ r1 r2, r1 = r2 → Rabs r1 = Rabs r2.

Lemma Req_dec : ∀ r1 r2, {r1 = r2} + {r1 ≠ r2}.

Lemma Req_or_neq : ∀ r, {r = 0} + {r ≠ 0}.

Lemma Rplus_eq_compat_r : ∀ r r1 r2, r1 = r2 → r1 + r = r2 + r.

Lemma Rmult_eq_compat_r : ∀ r1 r2 r, r1 = r2 → r1 × r = r2 × r.

Lemma Rmax_lt_lt_lt : ∀ x y z, Rmax x y < z ↔ x < z ∧ y < z.

Lemma Rmax_le_le_le : ∀ x y z, Rmax x y ≤ z ↔ x ≤ z ∧ y ≤ z.

Lemma Rmin_gt_gt_gt : ∀ x y z, Rmin x y > z ↔ x > z ∧ y > z.

Lemma Rmin_ge_ge_ge : ∀ x y z, Rmin x y ≥ z ↔ x ≥ z ∧ y ≥ z.

Lemma Rle_neq_lt : ∀ x y, x ≤ y ∧ x ≠ y → x < y.
