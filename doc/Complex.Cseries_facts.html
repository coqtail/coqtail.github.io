Lemma Cser_cv_re_compat Un l : Cser_cv Un l →
  Rser_cv (fun n : nat ⇒ Cre (Un n)) (Cre l).

Lemma Cser_cv_im_compat Un l : Cser_cv Un l →
  Rser_cv (fun n : nat ⇒ Cim (Un n)) (Cim l).

Lemma Cser_cv_re_im_compat : ∀ Un l,
  Rser_cv (fun n ⇒ Cre (Un n)) (Cre l) →
  Rser_cv (fun n ⇒ Cim (Un n)) (Cim l) →
  Cser_cv Un l.

Lemma Ccauchy_product : ∀ An Bn la lb lna,
 Cser_cv An la →
 Cser_cv Bn lb →
 Cser_abs_cv An lna →
 Cser_cv ((fun k:nat ⇒ sum_f_C0 (fun p:nat ⇒ An p × Bn (k - p)%nat) k)%C)
   (la × lb)%C.

Lemma Cser_cv_unique Un l1 l2 : Cser_cv Un l1 → Cser_cv Un l2 → l1 = l2.

Lemma Csum_eq_compat_weak Un Vn n :
  (∀ j, Un j = Vn j) → sum_f_C0 Un n = sum_f_C0 Vn n.

Lemma Csum_eq_compat Un Vn n :
  (∀ j, (j ≤ n)%nat → Un j = Vn j) → sum_f_C0 Un n = sum_f_C0 Vn n.

Lemma Cser_cv_eq_compat Un Vn l :
  (∀ j, Un j = Vn j) → Cser_cv Un l → Cser_cv Vn l.
