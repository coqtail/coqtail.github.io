Theorem R_uncountable_strong :
  ∀ (f : nat → R) (x y : R), x < y → {l : R | ∀ n, l ≠ f n & x ≤ l ≤ y}.

Theorem R_uncountable : ∀ (f : nat → R), {l : R | ∀ n, l ≠ f n}.
