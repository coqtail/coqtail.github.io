Definition odds Un : Rseq := fun n ⇒ Un (S (2 × n)).

Convergence of splitting
Lemma Rser_cv_pair_compat : ∀ Un l, Rser_cv Un l → Rser_cv (fun n ⇒ Un (mult 2 n) + Un (S (mult 2 n))) l.
Proof.
intros Un l Ucv.
intros e epos.
destruct (Ucv e epos) as [N Hu].
∃ N; intros n nN.
replace (Rseq_sum (fun n0 : nat ⇒ Un (2 × n0)%nat + Un (S (2 × n0))) n)
 with (Rseq_sum Un (S (2 × n))).
 intros; apply Hu; omega.

 clear nN.
 induction n.
  intros; simpl; ring.

  simpl Rseq_sum.
  replace (n + S (n + 0))%nat with (S (2 × n)) by ring.
  rewrite IHn.
  ring_simplify.
  assert (SIMPL : ∀ a b c d, a = c → a + b + d = b + d + c) by (intros; subst; ring).
  apply SIMPL.
  apply Rseq_sum_ext; intro.
  simpl; trivial.
Qed.

  sum_f_R0 an (S (2 × n)) - sum_f_R0 (evens an) n.

Qed.
