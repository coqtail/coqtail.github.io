Definition of a finite sum on nat with an index starting from 0


Fixpoint Nfinite_sum_0_n (n : nat) (f : nat → nat) {struct n} : nat :=
match n with
| 0 ⇒ f 0
| S n' ⇒ (f n) + Nfinite_sum_0_n n' f
end.
