Basic notions


Definition open_interval (lb ub x:R) := lb < x < ub.
Definition interval (lb ub x:R) := lb ≤ x ≤ ub.
Definition middle (x:R) (y:R) : R := (x+y)/2.

Definition continuity_open_interval (f : R → R) (lb ub:R) := ∀ x:R,
      open_interval lb ub x → continuity_pt f x.
Definition continuity_interval (f : R → R) (lb ub:R) := ∀ x:R,
      interval lb ub x → continuity_pt f x.

Definition derivable_open_interval (f : R → R) (lb ub:R) := ∀ x:R,
      open_interval lb ub x → derivable_pt f x.
Definition derivable_interval (f : R → R) (lb ub:R) := ∀ x:R,
      interval lb ub x → derivable_pt f x.

Definition injective_interval (f : R → R) (lb ub:R) := ∀ (x y:R),
      interval lb ub x → interval lb ub y → f x = f y → x = y.
Definition surjective_interval (f : R → R) (lb ub:R) := ∀ y:R,
      interval lb ub y → ∃ x:R, y = f x.

Definition injective (f : R → R) := ∀ (x y:R), f x = f y → x = y.
Definition surjective (f : R → R) := ∀ y:R, ∃ x:R, y = f x.
Definition bijective (f : R → R) := injective f ∧ surjective f.

Definition strictly_increasing_interval (f : R → R) (lb ub:R) := ∀ x y,
       interval lb ub x → interval lb ub y → x < y → f x < f y.
Definition strictly_decreasing_interval (f : R → R) (lb ub:R) := ∀ x y,
       interval lb ub x → interval lb ub y → x < y → f y < f x.
Definition strictly_monotonous_interval (f : R → R) (lb ub : R) :=
     {strictly_increasing_interval f lb ub} + {strictly_decreasing_interval f lb ub}.

Definition increasing_interval (f : R → R) (lb ub:R) := ∀ x y,
       interval lb ub x → interval lb ub y → x ≤ y → f x ≤ f y.
Definition decreasing_interval (f : R → R) (lb ub:R) := ∀ x y,
       interval lb ub x → interval lb ub y → x ≤ y → f y ≤ f x.
Definition monotonous_interval (f : R → R) (lb ub : R) :=
     {increasing_interval f lb ub} + {decreasing_interval f lb ub}.

Definition strictly_increasing (f : R → R) := ∀ x y, x < y → f x < f y.
Definition strictly_decreasing (f : R → R) := ∀ x y, x < y → f y < f x.
Definition strictly_monotonous (f : R → R) :=
     {strictly_increasing f} + {strictly_decreasing f}.

Definition reciprocal_interval (f g:R → R) (lb ub:R) := ∀ x,
       interval lb ub x → (comp f g) x = id x.
Definition reciprocal_open_interval (f g:R → R) (lb ub:R) := ∀ x,
       open_interval lb ub x → (comp f g) x = id x.

Definition reciprocal (f g:R → R) := ∀ x, (comp f g) x = id x.
